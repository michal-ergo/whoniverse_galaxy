	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_playable_country = {
			num_unique_species > 1
			is_whoniverse_dalek = yes
		}
	}
	
	immediate = {
		every_playable_country = {
			limit = {
				num_unique_species > 1
				is_whoniverse_dalek = yes
			}

				planet_event = { id = whoniverse.5 }
			}
		}
	}

# Create new Pops on Yearly Pulse (Dalek-Conversion)
planet_event = {
	id = whoniverse.5
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_owned_species = {
			NOR = {
				is_same_species = root.owner
				is_robotic_species = yes
				is_mechanical_species = yes
				is_whoniverse_human_species = no
			}
		}
	}
	
	immediate = {
		random_list = {
			1 = { # 1 pop
				create_pop = {
					species = root.owner
					ethos = root.owner
				}
			}
			1 = { # 2 pop
				while = {
					count = 2
					create_pop = {
						species = root.owner
						ethos = root.owner
					}
				}
			}
			1 = { # 3 pop
				while = {
					count = 3
					create_pop = {
						species = root.owner
						ethos = root.owner
					}
				}
			}
		}
	}
}

		has_owner = yes
		is_homeworld = no
		has_ground_combat = no
		is_ai = yes
		is_capital = no
	}

	trigger = {
		from = {
			is_whoniverse_dalek = yes
			is_ai = yes
			}

		num_pops > 0
		}

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				free_building_slots <= 0
				has_district = district_generator

				}
				remove_district = district_generator
				add_district = district_city
				}
		add_building = building_whoniverse_dalek_incubation_vats
		set_planet_flag = can_organic_assemble_flag
		 	}
	}
