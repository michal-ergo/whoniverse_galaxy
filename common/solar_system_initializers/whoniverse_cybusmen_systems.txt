	class = "sc_black_hole"

	usage = misc_system_init
	usage_odds = {
		base = 0
		modifier = {
			add = 10000
			NOR = {
				any_country = { has_country_flag = whoniverse_cybusmen }
				any_system = { has_star_flag = whoniverse_cybusmen_start_system }
			}
		}
	}
	max_instances = 1

	asteroid_belt = {
		type = rocky_asteroid_belt
		radius = 160
	}

	planet = {
		name = "The Void"
		class = "pc_black_hole"
		orbit_distance = 0
		orbit_angle = 1
		size = 60
		has_ring = no
	}

	init_effect = {
		set_spawn_system_batch = begin
		spawn_system = {
			max_distance = 180
			min_distance = 130
			max_jumps = 1
			min_jumps = 1
			direction = rimwards
			initializer = whoniverse_cybusmen_initializer
			hyperlane = no
			effect = {
				save_event_target_as = cybus_system
			}
		}
		event_target:cybus_system = {
			prev = {
				spawn_natural_wormhole = {
					bypass_type = wormhole
					random_pos = no
					orbit_angle = 90
				}
			}
			spawn_natural_wormhole = {
				bypass_type = wormhole
				random_pos = yes
				orbit_angle = 90
			}
			link_wormholes = prev
		}
		set_spawn_system_batch = end
	}

}
