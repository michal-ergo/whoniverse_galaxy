#Dalek Ascension - enables creation of puppet armies
ap_whoniverse_dalek = {
	on_enabled = {
		custom_tooltip = "whoniverse_ap_dalek_tooltip"
	}
	possible = {
		host_has_dlc = "Necroids Species Pack"
		custom_tooltip = {
			fail_text = "requires_free_tradition_tree"
			num_tradition_categories < @max_tradition_trees
		}
	}

	potential = {
		is_whoniverse_dalek = yes
	}

	ai_weight = {
		factor = 100
	}
}
